DARTWARE-MIB DEFINITIONS ::= BEGIN

    IMPORTS
        enterprises
            FROM RFC1155-SMI
        DisplayString
            FROM RFC1213-MIB
        OBJECT-TYPE
            FROM RFC-1212
        TRAP-TYPE
            FROM RFC-1215;
     
    -- Define SNMPv1 Traps sent by InterMapper 3.0 and later
	-- (23 October 2000)
	
	-- Updated all descriptions;
	-- Added Critical event type to intermapperMessage; 
	-- Added intermapperSender and intermapperProbe variables.
	-- (9 May 2007)

    dartware        OBJECT IDENTIFIER ::= { enterprises 6306 }
    notify          OBJECT IDENTIFIER ::= { dartware 2 }
    intermapper     OBJECT IDENTIFIER ::= { notify 1 }
    
    intermapperTimestamp OBJECT-TYPE
        SYNTAX      DisplayString (SIZE(0..255))
        ACCESS      read-only
        STATUS      mandatory
        DESCRIPTION
			"The current date and time, as a string in the format 'mm/dd hh:mm:ss'."
        ::= { intermapper 1 }
        
    intermapperMessage OBJECT-TYPE
        SYNTAX      DisplayString (SIZE(0..255))
        ACCESS      read-only
        STATUS      mandatory
        DESCRIPTION
			"The type of event: Down, Up, Critical, Alarm, Warning, OK, or Trap, as a string."
        ::= { intermapper 2 }

    intermapperDeviceName OBJECT-TYPE
        SYNTAX      DisplayString (SIZE(0..255))
        ACCESS      read-only
        STATUS      mandatory
        DESCRIPTION
			"The (first line of the) label of the device as shown on a map, as a string."
        ::= { intermapper 3 }

    intermapperCondition OBJECT-TYPE
        SYNTAX      DisplayString (SIZE(0..255))
        ACCESS      read-only
        STATUS      mandatory
        DESCRIPTION
            "The condition of the device, as it would be printed in the Event Log file."
        ::= { intermapper 4 }

	intermapperSender OBJECT-TYPE
		SYNTAX 	DisplayString (SIZE(0..255))
		ACCESS		read-only
		STATUS 		mandatory
		DESCRIPTION
			"The IP address of the trap's sender."
		::= { intermapper 5 }

	intermapperProbe OBJECT-TYPE
		SYNTAX 	DisplayString (SIZE(0..255))
		ACCESS		read-only
		STATUS 		mandatory
		DESCRIPTION
			"The probe that sent the trap."
		::= { intermapper 6 }

    intermapperTrap TRAP-TYPE
        ENTERPRISE  dartware
		VARIABLES	{ intermapperTimestamp, intermapperMessage, intermapperDeviceName,
		intermapperCondition, intermapperSender, intermapperProbe }
        DESCRIPTION
            "The SNMP trap that is generated by InterMapper as a notification option."
        ::= 1

END
